describe("nodeJS services", ()=>{
    it ("should handle authentication with OAuth2.0")
    it ("should only allow access to most pages with valid access token")
    it ("should only allow access to service with valid access token")
    
    describe ("authentication services", ()=>{

        describe ("Sign up service", ()=>{
            it ("should accept a `username` string and `password` string")
            it ("should reject if `username` or `password` is empty")
            it ("should reject if `username` is already taken by previous user")
            it ("should add new row in DB")
            describe ("successful sign up", ()=>{
                it ("should return status 200")
            })
            describe ("failed sign up", ()=>{
                it ("should return error status")
                it ("should return error message")
            })
        })

        describe ("Sign in service", ()=>{
            it ("should accept a `username` string and `password` string")
            it ("should reject if `username` or `password` is empty")
            it ("should process `password` with bcrypt")
            it ("should reject if `username` is not found in DB")
            it ("should reject if stored `password` is not equal processed `password`")
            describe ("successful sign up", ()=>{
                it ("should update `updated_at`, `is_signed_in`, `last_sign_in`")
                it ("should return status 200")
            })
            describe ("failed login", ()=>{
                it ("should return error status")
                it ("should return error message")
            })
        })

        describe ("Facebook sign in service", ()=>{
            it ("should accept a token generated by Facebook")
            it ("should call Facebook API to fetch user info")
            describe ("failed API call", ()=>{
                it ("should return error status")
                it ("should return error message: Facebook service not available")
            })
            describe ("successful API call", ()=>{
                it ("should look for `socialID` in DB")
                describe ("socialID is found in DB", ()=>{
                    it ("should update `updated_at`, `is_signed_in`, `last_sign_in`")
                    it ("should set user session")
                    it ("should return status 200")
                })
                describe ("socialID not found in DB", ()=>{
                    it ("should add new row in DB")
                    it ("should set user session")
                    it ("should return status 200")
                })
                describe ("DB write error", ()=>{
                    it ("should return error status")
                    it ("should return error message")
                })
            })
        })
    })

    describe ("page serving", ()=>{
        it ("should serve `landing` page to every other protected requests")
        it ("should serve `lobby` and `game` page only to authenticated request")
        
        describe ("landing page", ()=>{
            it ("should always just serve the page")
        })

        describe ("lobby page", ()=>{
            it ("should check session before serving page")
        })

        describe ("game page", ()=>{
            it ("should check session before serving page")
        })
    })
})
